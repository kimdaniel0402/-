    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, onSnapshot, setDoc, doc, updateDoc, deleteDoc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        
        // --- 전역 변수 설정 (Canvas 환경 제공) ---
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : '';
        const apiKey = ""; // API Key (Placeholder)
        
        // --- Firebase 초기화 및 인증 ---
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        let userId = 'loading'; // 초기 사용자 ID
        
        // Firestore 컬렉션 경로 (모두가 공유할 수 있도록 Public 경로 사용)
        const ENTRIES_COLLECTION_PATH = `artifacts/${appId}/public/data/ledger_entries`;

        window.firebaseReady = new Promise(async (resolve) => {
            try {
                // 커스텀 토큰이 제공되면 로그인, 아니면 익명 로그인
                if (initialAuthToken) {
                    await signInWithCustomToken(auth, initialAuthToken);
                } else {
                    await signInAnonymously(auth);
                }
                
                userId = auth.currentUser?.uid || crypto.randomUUID();
                console.log('Firebase initialized. User ID:', userId);

                // 가계부 동기화 시작
                window.startSync(db);
                
                resolve(true);

            } catch (error) {
                console.error("Firebase Auth or Init Error:", error);
                document.getElementById('sync-status').textContent = '데이터베이스 연결 실패. 콘솔을 확인하세요.';
                document.getElementById('sync-status').classList.replace('text-yellow-600', 'text-red-600');
                resolve(false);
            }
        });
    </script>

    <script type="module">
        // Firebase 관련 모듈들을 window.moduleExports에 할당하여 아래 <script>에서 접근 가능하게 함
        import { getFirestore, collection, onSnapshot, setDoc, doc, updateDoc, deleteDoc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        
        window.moduleExports = { getFirestore, collection, onSnapshot, setDoc, doc, updateDoc, deleteDoc, query };
    </script>

    <script>
        // --- 전역 데이터 및 UI 요소 ---
        let generalLedger = []; 
        const categoryNames = ['식사', '유흥', '쇼핑', '월정액'];
        const categories = {}; // 카테고리별 집계 데이터
        const glList = document.getElementById('general-ledger-list');
        const totalSumDisplay = document.getElementById('total-sum');
        const glForm = document.getElementById('general-ledger-form');
        const categoryContainer = document.getElementById('category-container');
        const submitBtn = document.getElementById('submit-btn');

        // Firestore 전역 변수 (startSync에서 할당됨)
        let firestoreDb;
        let entriesCollectionRef;
        const ENTRIES_COLLECTION_PATH = `artifacts/${typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'}/public/data/ledger_entries`;
        let userId = 'loading'; // 초기 사용자 ID
        
        // --- 유틸리티 함수 ---

        function formatCurrency(amount) {
            if (isNaN(amount) || amount === null) return '0 원';
            return amount.toLocaleString('ko-KR') + ' 원';
        }

        // --- Firebase 동기화 로직 ---

        window.startSync = (db) => {
            firestoreDb = db;
            entriesCollectionRef = moduleExports.collection(firestoreDb, ENTRIES_COLLECTION_PATH);
            
            // 사용자 ID 표시
            userId = window.auth?.currentUser?.uid || 'anonymous';
            document.getElementById('user-id').textContent = userId;

            // 실시간 리스너 설정
            moduleExports.onSnapshot(moduleExports.query(entriesCollectionRef), (snapshot) => {
                generalLedger = [];
                snapshot.forEach(doc => {
                    // Firestore 문서 ID를 기록 데이터에 추가하여 삭제 시 사용
                    generalLedger.push({ id: doc.id, ...doc.data() });
                });
                console.log('Data synced from Firestore. Total entries:', generalLedger.length);
                
                recalculateAndRenderAll();
                
                document.getElementById('sync-status').textContent = '실시간 동기화 완료';
                document.getElementById('sync-status').classList.replace('text-yellow-500', 'text-green-500');
            }, (error) => {
                console.error("Firestore Snapshot Error:", error);
                document.getElementById('sync-status').textContent = '동기화 오류 발생. 콘솔 확인.';
                document.getElementById('sync-status').classList.replace('text-green-500', 'text-red-500');
            });

            // 데이터베이스 준비 완료 후 입력 활성화
            submitBtn.disabled = false;
            submitBtn.textContent = '지출 등록 및 동기화';
        };

        // --- 렌더링 및 계산 로직 ---

        function recalculateAndRenderAll() {
            // 1. 카테고리 데이터 초기화 및 재분류 (Section 2 데이터 준비)
            categoryNames.forEach(name => {
                categories[name] = [];
            });

            generalLedger.forEach(entry => {
                if (categories[entry.category]) {
                    categories[entry.category].push({
                        title: entry.item,
                        amount: entry.amount,
                        date: entry.date,
                        id: entry.id // 삭제를 위해 ID 유지
                    });
                }
            });

            // 2. 모든 UI 업데이트
            renderGeneralLedger();
            renderAllCategories();
        }

        function renderGeneralLedger() {
            glList.innerHTML = ''; 

            if (generalLedger.length === 0) {
                glList.innerHTML = '<li class="text-center text-sm text-gray-400 py-4">기록된 지출이 없습니다.</li>';
                calculateTotalSum();
                return;
            }

            // 날짜 최신순으로 정렬
            generalLedger.sort((a, b) => new Date(b.date) - new Date(a.date));

            generalLedger.forEach((entry) => {
                const li = document.createElement('li');
                li.className = 'flex justify-between items-center bg-gray-50 p-3 rounded-lg border border-gray-200 hover:bg-red-100 transition duration-150 cursor-pointer';
                li.setAttribute('data-id', entry.id);
                li.title = '클릭하면 이 지출 내역을 삭제합니다.';

                li.innerHTML = `
                    <div class="flex-1 min-w-0">
                        <p class="text-xs font-semibold text-gray-500">${entry.date} / ${entry.category}</p> 
                        <p class="text-sm font-medium text-gray-800 truncate">${entry.item}</p>
                    </div>
                    <div class="text-right ml-4">
                        <p class="text-md font-bold text-red-600">${formatCurrency(entry.amount)}</p>
                    </div>
                `;
                glList.appendChild(li);
            });
            calculateTotalSum();
        }

        function calculateTotalSum() {
            const total = generalLedger.reduce((sum, entry) => sum + entry.amount, 0);
            totalSumDisplay.textContent = formatCurrency(total);
        }

        /**
         * 특정 카테고리의 총 합계를 계산합니다. (오류 수정으로 추가됨)
         * @param {string} category - 카테고리 이름
         * @returns {number} - 해당 카테고리의 총 금액
         */
        function calculateCategoryTotal(category) {
            return categories[category].reduce((sum, entry) => sum + entry.amount, 0);
        }


        function renderCategory(name) {
            const card = document.getElementById(`card-${name}`);
            if (!card) return;

            const totalDisplay = card.querySelector(`#total-${name}`);
            const list = card.querySelector(`#list-${name}`);
            const total = calculateCategoryTotal(name);

            totalDisplay.textContent = formatCurrency(total);
            list.innerHTML = ''; 

            if (categories[name].length === 0) {
                list.innerHTML = '<li class="text-center text-xs text-gray-400 py-2">지출 내역이 없습니다.</li>';
                return;
            }

            categories[name].forEach((entry) => {
                const li = document.createElement('li');
                li.className = 'flex justify-between items-center bg-white p-2 text-sm border-b border-gray-100 last:border-b-0';
                li.innerHTML = `
                    <span class="text-gray-700 truncate">${entry.title}</span>
                    <span class="font-semibold text-red-500">${formatCurrency(entry.amount)}</span>
                `;
                list.appendChild(li);
            });
        }

        function renderAllCategories() {
            categoryNames.forEach(name => renderCategory(name));
        }

        // --- 이벤트 핸들러 ---

        glForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            const dateInput = document.getElementById('gl-date');
            const itemInput = document.getElementById('gl-item');
            const amountInput = document.getElementById('gl-amount');
            const categoryInput = document.getElementById('gl-category'); 

            const newEntry = {
                date: dateInput.value,
                item: itemInput.value.trim(),
                amount: parseInt(amountInput.value, 10) || 0,
                category: categoryInput.value, 
                // 기록 시점의 타임스탬프를 추가 (정렬 및 생성 시점 기록)
                createdAt: new Date().toISOString() 
            };

            if (newEntry.amount > 0 && newEntry.item && newEntry.category) {
                submitBtn.disabled = true;
                submitBtn.textContent = '저장 중...';

                try {
                    // Firestore에 새로운 문서로 추가 (ID는 Firestore가 자동 생성)
                    await moduleExports.setDoc(moduleExports.doc(entriesCollectionRef), newEntry);
                    
                    // 성공 시 폼 초기화
                    itemInput.value = '';
                    amountInput.value = '';
                    dateInput.value = new Date().toISOString().split('T')[0];
                } catch (error) {
                    console.error("Error adding document: ", error);
                    document.getElementById('sync-status').textContent = '저장 실패!';
                    setTimeout(() => { document.getElementById('sync-status').textContent = '실시간 동기화 완료'; }, 3000);
                } finally {
                    submitBtn.disabled = false;
                    submitBtn.textContent = '지출 등록 및 동기화';
                }
            }
        });

        // 지출 기록 삭제 이벤트
        glList.addEventListener('click', async (e) => {
            const listItem = e.target.closest('li[data-id]');
            if (!listItem) return;
            
            // 커스텀 모달 또는 확인 로직을 구현해야 하지만, 편의를 위해 바로 삭제
            const entryId = listItem.getAttribute('data-id');
            const confirmed = window.confirm('정말로 이 기록을 삭제하시겠습니까?');

            if (confirmed) {
                try {
                    const entryRef = moduleExports.doc(entriesCollectionRef, entryId);
                    await moduleExports.deleteDoc(entryRef);
                    // 삭제 성공 후 onSnapshot이 자동으로 UI 업데이트를 담당함
                    console.log(`Entry ${entryId} deleted.`);
                } catch (error) {
                    console.error("Error removing document: ", error);
                    document.getElementById('sync-status').textContent = '삭제 실패!';
                }
            }
        });

        // --- 초기 설정 ---

        function setupCategoryCards() {
            categoryContainer.innerHTML = ''; 
            categoryNames.forEach(name => {
                const card = document.createElement('div');
                card.id = `card-${name}`;
                card.className = 'bg-white p-5 rounded-xl shadow-md border border-gray-200';
                
                const colorMap = {
                    '식사': 'bg-green-50 text-green-700 border-green-300',
                    '유흥': 'bg-red-50 text-red-700 border-red-300',
                    '쇼핑': 'bg-blue-50 text-blue-700 border-blue-300',
                    '월정액': 'bg-purple-50 text-purple-700 border-purple-300'
                };
                const color = colorMap[name] || 'bg-gray-50 text-gray-700 border-gray-300';

                card.innerHTML = `
                    <div class="flex items-center justify-between ${color} p-3 rounded-lg mb-4">
                        <h3 class="text-xl font-bold">${name}</h3>
                        <div class="text-sm font-medium">합계:</div>
                    </div>
                    
                    <div class="mb-4 text-center p-2 border-b">
                        <p id="total-${name}" class="text-2xl font-extrabold text-gray-800">0 원</p>
                    </div>

                    <h4 class="text-md font-semibold text-gray-700 mt-4 mb-2 border-t pt-2">내역</h4>
                    <ul id="list-${name}" class="space-y-1 max-h-32 overflow-y-auto scroll-list bg-gray-50 rounded-lg p-2">
                        <li class="text-center text-xs text-gray-400 py-2">지출 내역이 없습니다.</li>
                    </ul>
                `;
                categoryContainer.appendChild(card);
            });
        }

        window.onload = function() {
            // 초기 날짜 설정
            document.getElementById('gl-date').value = new Date().toISOString().split('T')[0];
            
            // 카테고리 카드 UI 설정
            setupCategoryCards();
            
            // Firebase 초기화가 완료되면 window.startSync가 호출되어 데이터 동기화 시작
        };

    </script>
